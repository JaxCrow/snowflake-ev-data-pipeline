  GCS -->|"JSON files"| STAGE
  PUBSUB -->|"File event"| STAGE
  STAGE --> DIR_STREAM
  DIR_STREAM -->|"WHEN has_data"| INGEST_TASK
  INGEST_TASK --> INGEST_SP
  INGEST_SP -->|"COPY INTO"| RAW

  RAW -->|"Auto-detect"| DT
  DT --> DT_STREAM

  DT_STREAM -->|"WHEN has_data"| GOLD_TASK
  GOLD_TASK --> GOLD_SP
  GOLD_SP -->|"Enrich"| FACT_REG
  GOLD_SP -->|"Aggregate"| FACT_ICE
  POSTGRES -->|"CDC"| DIM_MFG

  FACT_REG --> SEMANTIC_VIEW
  DIM_MFG --> SEMANTIC_VIEW
  SEMANTIC_VIEW --> CORTEX
  CORTEX -->|"SQL"| STREAMLIT

  FACT_ICE --> SV_METRICS
  DIM_MFG --> SV_DIM
  SV_METRICS --> SHARE
  SV_DIM --> SHARE
  SV_DQ --> SHARE
  SHARE --> READER
  FACT_ICE --> CATALOG_SYNC
  CATALOG_SYNC --> EXT_ENGINES

  RAW -.->|"Monitor"| DMF
